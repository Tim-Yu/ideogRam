set.seed(1)
genes <- genes(EnsDb.Hsapiens.v75)
genes <- genes[start(genes) < 40000000]
genes <- genes[sample(seq(length(genes)), 100)]

p <- ideogRam(organism = "human") %>%
    set_option(orientation = "horizontal") %>%
    add_track(genes)
p
